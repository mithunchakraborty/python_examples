$ pip install django djangorestframework django-cors-headers
$ django-admin startproject restreact .
$ python3 manage.py startapp customers

$ npx create-react-app frontend

Запуск:
$ cd ~/restreact/frontend
$ npm start

Axios:
$ cd ~/restreact/frontend
$ npm install axios --save

Пустой файл миграции:
$ python3 manage.py makemigrations --empty --name customers customers
